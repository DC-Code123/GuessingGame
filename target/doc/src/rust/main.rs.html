1//! Main entry point for the number guessing game
2//! 
3//! This implements a complete guessing game where:
4//! - A random number between 1.0-100.0 is generated
5//! - Players guess with optional hints
6//! - Supports retrying with same or new numbers
7
8use std::io;
9use std::process::exit;
10use crate::utils::{game_loop, gen_rand, end_situation_handler, choose_hint};
11mod utils;
12
13fn main() {
14    // Print game introduction
15    println!("Welcome to the Fantastic Number Guessing Game!");
16    println!("=============================================");
17    println!("A random number between 1.0 and 100.0 will be generated.");
18    println!("Try to guess it with optional hints to help you!\n");
19
20    // Main game loop - runs until player chooses to quit
21    'game: loop {
22        // Generate new secret number for each game session
23        let secret_number = gen_rand();
24        let mut attempts = 0;
25        println!("\nNew game started! A secret number has been generated.");
26
27        // Retry loop - allows playing same number multiple times
28        'retry: loop {
29            // Get player's hint preference
30            let hint_choice = get_hint_choice();
31            
32            // Show selected hint type
33            choose_hint(&hint_choice, secret_number);
34
35            // Run one full game round and get results
36            let (guess_correct, new_attempts) = game_loop(secret_number, attempts);
37            attempts = new_attempts;
38
39            // Handle post-game choices
40            match end_situation_handler(guess_correct, attempts) {
41                1 => { // Player wants to continue
42                    match get_retry_choice() {
43                        1 => { // Retry same number
44                            println!("\nContinuing with same number. Attempts reset.");
45                            continue 'retry;
46                        },
47                        2 => { // Get new number
48                            println!("\nGenerating new number...");
49                            continue 'game;
50                        },
51                        0 => { // Quit game
52                            println!("\nGame over in {} attempts. -quiting", attempts);
53                            exit_game();
54                        },
55                        _ => { // Invalid input
56                            println!("Invalid choice. Starting new game.");
57                            continue 'game;
58                        }
59                    }
60                },
61                0 => { // Player chose to quit
62                    exit_game();
63                },
64                _ => { // Error case
65                    println!("Unexpected error. Exiting.");
66                    exit(1);
67                }
68            }
69        }
70    }
71}

73/// Prompts player to select hint type
74/// Returns:
75///   String containing their choice ("1", "2", or "3")
76fn get_hint_choice() -> String {
77    println!("\nChoose a hint option:");
78    println!("1. Easy hint (simple arithmetic)");
79    println!("2. Hard hint (complex equations)");
80    println!("3. No hints (I'm feeling lucky!)");
81    println!("Your choice (1-3, default 3): ");
82    
83    let mut choice = String::new();
84    io::stdin().read_line(&mut choice).expect("Failed to read input");
85    choice
86}
87
88/// Gets player's choice after game ends
89/// Returns:
90///   1 = same number, 2 = new number, 0 = quit
91fn get_retry_choice() -> i32 {
92    println!("\nWhat would you like to do next?");
93    println!("1. Try same number again");
94    println!("2. Get a new random number");
95    println!("0. Quit game");
96    println!("Your choice (0-2): ");
97    
98    let mut choice = String::new();
99    io::stdin().read_line(&mut choice).expect("Failed to read input");
100    choice.trim().parse().unwrap_or(0) // Default to 0 (quit) on invalid input
101}

103/// Cleanly exits the game with farewell message
104fn exit_game() {
105    println!("\nThank you for playing! Goodbye!");
106    exit(0);
